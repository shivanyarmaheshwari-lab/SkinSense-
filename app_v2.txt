# Extract unique options for dropdowns, removing duplicates and sorting
appearances = sorted(list(df_diseases["Appearance"].unique()))

durations = sorted(list(df_diseases["Duration"].unique()))

appearances.insert(0, "Select an option...")

# Patient Information Section
st.markdown("### Patient Profile")
col1, col2, col3 = st.columns(3)
with col1:
    patient_name = st.text_input("Name")
with col2:
    patient_age = st.number_input("Age", min_value=0, max_value=120, step=1)
with col3:
    patient_gender = st.selectbox("Gender", ["Select...", "Male", "Female", "Other", "Prefer not to say"])

st.divider()

# Image Upload Section
st.markdown("### Visual Reference")
upload_choice = st.checkbox("I would like to upload an image of my skin condition (Optional)")

if upload_choice:
    uploaded_file = st.file_uploader("Choose an image...", type=["jpg", "jpeg", "png"])

    if uploaded_file is not None:
        # Calculate file size in KB
        file_size_kb = uploaded_file.size / 1024

        # Display the image
        img = Image.open(uploaded_file)
        st.image(img, caption="Uploaded Image Reference", use_container_width=True)

        # Display the size
        st.info(f"📁 **Image Size:** {file_size_kb:.2f} KB")

st.divider()

# Symptoms Section
st.markdown("### Symptom Details")
st.write("Please select the closest match for each symptom category. You can leave fields as 'Select an option...' if you are unsure.")

selected_appearance = st.selectbox("Appearance:", appearances)
selected_sensation = st.selectbox("Sensation:", sensations)
selected_location = st.selectbox("Location:", locations)
selected_duration = st.selectbox("Duration:", durations)
selected_colour = st.selectbox("Colour:", colours)

st.divider()

# --- MATCHING LOGIC ---
if st.button("Find Matching Condition", type="primary"):

    # Check if patient name is entered for personalized output
    greeting = f"Here are the findings for **{patient_name}**:" if patient_name else "Here are the findings:"
    st.markdown(f"#### {greeting}")

    # Calculate match scores
    results = []

    for index, row in df_diseases.iterrows():
        score = 0
        total_selected = 0

        if selected_appearance != "Select an option...":
            total_selected += 1
            if row["Appearance"] == selected_appearance: score += 1

        if selected_sensation != "Select an option...":
            total_selected += 1
            if row["Sensation"] == selected_sensation: score += 1

        if selected_location != "Select an option...":
            total_selected += 1
            if row["Location"] == selected_location: score += 1

        if selected_duration != "Select an option...":
            total_selected += 1
            if row["Duration"] == selected_duration: score += 1

        if selected_colour != "Select an option...":
            total_selected += 1
            if row["Colour"] == selected_colour: score += 1

        if total_selected > 0:
            match_percentage = (score / total_selected) * 100
        else:
            match_percentage = 0

        results.append({
            "Condition": row["Condition"],
            "Matches": score,
            "Match Percentage": match_percentage
        })

    # Sort results by score
    results_df = pd.DataFrame(results)
    results_df = results_df.sort_values(by=["Matches", "Match Percentage"], ascending=[False, False])

    # Filter out 0 matches if the user selected at least something
    if sum([1 for x in [selected_appearance, selected_sensation, selected_location, selected_duration, selected_colour] if x != "Select an option..."]) > 0:
        top_matches = results_df[results_df["Matches"] > 0]

        if not top_matches.empty:
            st.success("Potential matches found based on your selections:")

            # Display top 3 results
            for i, row in top_matches.head(3).iterrows():
                st.markdown(f"- **{row['Condition']}** (Matched {row['Matches']} symptom(s))")

            st.info("Remember, dermatological conditions often overlap in symptoms. Only a doctor can provide a definitive diagnosis.")
        else:
            st.error("No exact symptom matches found in the database. Please try adjusting your selections or consult a physician.")
    else:
        st.warning("Please select at least one symptom from the dropdowns above to see a match.")
